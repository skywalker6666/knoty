- [ ] **Step 1: Create the helper**

```typescript
// apps/web/lib/supabase-server.ts
import { cookies } from 'next/headers';
import { createServerClient } from '@knoty/api-client';

/**
 * Creates a Supabase client for API Routes and Server Components.
 * Uses @supabase/ssr cookie store so auth.uid() resolves correctly via RLS.
 * Always use this instead of createAdminClient() — the admin client bypasses RLS.
 */
export async function createRouteClient() {
  const cookieStore = await cookies();
  return createServerClient({
    getAll: () => cookieStore.getAll(),
    setAll: (cs) =>
      cs.forEach(({ name, value, options }) =>
        cookieStore.set(name, value, options),
      ),
  });
}

/**
 * Auth check helper for API routes.
 * Returns the authenticated user or null.
 */
export async function getAuthUser(supabase: Awaited<ReturnType<typeof createRouteClient>>) {
  const { data: { user } } = await supabase.auth.getUser();
  return user;
}
```

```typescript
// apps/web/middleware.ts
import { NextRequest, NextResponse } from 'next/server';
import { createServerClient } from '@knoty/api-client';

// All /api/* routes handle their own auth and return JSON 401 — do NOT redirect them.
// Only redirect browser navigations (non-API routes).
const PUBLIC_PATHS = ['/login', '/auth/callback', '/api/'];

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl;

  // Allow public paths through
  if (PUBLIC_PATHS.some((p) => pathname.startsWith(p))) {
    return NextResponse.next();
  }

  const response = NextResponse.next();

  // createServerClient with cookie read/write for session refresh
  const supabase = createServerClient({
    getAll: () => request.cookies.getAll(),
    setAll: (cs) =>
      cs.forEach(({ name, value, options }) =>
        response.cookies.set(name, value, options),
      ),
  });

  // Refresh session (required by @supabase/ssr)
  const { data: { user } } = await supabase.auth.getUser();

  if (!user) {
    const loginUrl = new URL('/login', request.url);
    return NextResponse.redirect(loginUrl);
  }

  return response;
}

export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico).*)',
  ],
};
```

```typescript
// apps/web/app/auth/callback/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { createRouteClient } from '@/lib/supabase-server';

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  const code = searchParams.get('code');
  const siteUrl = process.env.NEXT_PUBLIC_SITE_URL ?? 'http://localhost:3000';

  if (!code) {
    return NextResponse.redirect(new URL('/login', siteUrl));
  }

  const supabase = await createRouteClient();
  const { error } = await supabase.auth.exchangeCodeForSession(code);

  if (error) {
    return NextResponse.redirect(new URL('/login?error=auth', siteUrl));
  }

  // Check if new user (no persons yet)
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) {
    return NextResponse.redirect(new URL('/login', siteUrl));
  }

  const { count } = await supabase
    .from('persons')
    .select('*', { count: 'exact', head: true })
    .eq('user_id', user.id);

  // count is null on DB error — treat as returning user to avoid redirect loop
  const destination = count === 0 ? '/onboarding' : '/';
  return NextResponse.redirect(new URL(destination, siteUrl));
}
```

Create `apps/web/app/api/persons/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { z } from 'zod';
import { createRouteClient, getAuthUser } from '@/lib/supabase-server';

const CreatePersonSchema = z.object({
  display_name: z.string().min(1).max(100),
  avatar_emoji: z.string().max(10).optional(),
  circles: z.array(z.string()).optional().default([]),
  tags: z.array(z.string()).optional().default([]),
  notes: z.string().max(500).optional(),
});

export async function GET(_req: NextRequest) {
  const supabase = await createRouteClient();
  const user = await getAuthUser(supabase);
  if (!user) {
    return NextResponse.json({ error: 'Unauthorized', code: 'AUTH_REQUIRED' }, { status: 401 });
  }

  const { data, error } = await supabase
    .from('persons')
    .select('id, display_name, avatar_emoji, circles, tags, notes, created_at, updated_at')
    .eq('user_id', user.id)
    .order('display_name');

  if (error) {
    return NextResponse.json({ error: error.message, code: 'DB_ERROR' }, { status: 500 });
  }

  return NextResponse.json({ persons: data ?? [] });
}

export async function POST(req: NextRequest) {
  const supabase = await createRouteClient();
  const user = await getAuthUser(supabase);
  if (!user) {
    return NextResponse.json({ error: 'Unauthorized', code: 'AUTH_REQUIRED' }, { status: 401 });
  }

  const body: unknown = await req.json();
  const parsed = CreatePersonSchema.safeParse(body);
  if (!parsed.success) {
    return NextResponse.json({ error: parsed.error.message, code: 'VALIDATION_ERROR' }, { status: 400 });
  }

  const { data, error } = await supabase
    .from('persons')
    .insert({ ...parsed.data, user_id: user.id })
    .select()
```

```typescript
  const body: unknown = await req.json();
```

```typescript
  }

  let { person_a, person_b, direction } = parsed.data;

  if (person_a === person_b) {
    return NextResponse.json({ error: '不能與自己建立關係', code: 'VALIDATION_ERROR' }, { status: 400 });

  // Enforce person_a < person_b (DB CHECK constraint)
  if (person_a > person_b) {
    [person_a, person_b] = [person_b, person_a];
    // Flip direction
    if (direction === 'a_to_b') direction = 'b_to_a';
    else if (direction === 'b_to_a') direction = 'a_to_b';
  }

  const { data, error } = await supabase
    .from('relationships')
    .insert({ ...parsed.data, person_a, person_b, direction, user_id: user.id })
    .select()
    .single();
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { z } from 'zod';
import { createRouteClient, getAuthUser } from '@/lib/supabase-server';

const OnboardingSchema = z.object({
  persons: z.array(z.object({
    display_name: z.string().min(1).max(100),
    circles: z.array(z.string()).optional().default([]),
  })).min(1),
  relationships: z.array(z.object({
    person_a_idx: z.number().int().min(0),
    person_b_idx: z.number().int().min(0),
    closeness: z.number().int().min(1).max(5),
  })),
});

export async function POST(req: NextRequest) {
  const supabase = await createRouteClient();
  const user = await getAuthUser(supabase);
  if (!user) return NextResponse.json({ error: 'Unauthorized', code: 'AUTH_REQUIRED' }, { status: 401 });

  const body: unknown = await req.json();
  const parsed = OnboardingSchema.safeParse(body);
  if (!parsed.success) return NextResponse.json({ error: parsed.error.message, code: 'VALIDATION_ERROR' }, { status: 400 });

  const { persons: personInputs, relationships: relInputs } = parsed.data;

  // Batch insert persons
  const { data: createdPersons, error: personError } = await supabase
    .from('persons')
    .insert(personInputs.map((p) => ({ ...p, user_id: user.id })))
    .select('id');

  if (personError || !createdPersons) {
    return NextResponse.json({ error: personError?.message ?? 'Insert failed', code: 'DB_ERROR' }, { status: 500 });
  }

  // Build relationships using generated UUIDs
  const relInserts = relInputs
    .filter((r) => r.person_a_idx !== r.person_b_idx && r.person_a_idx < createdPersons.length && r.person_b_idx < createdPersons.length)
    .map((r) => {
      let pA = createdPersons[r.person_a_idx].id;
      let pB = createdPersons[r.person_b_idx].id;
      if (pA > pB) [pA, pB] = [pB, pA]; // enforce a < b
      return { person_a: pA, person_b: pB, closeness: r.closeness, direction: 'mutual', user_id: user.id };
    });

  if (relInserts.length > 0) {
    const { error: relError } = await supabase.from('relationships').insert(relInserts);
    if (relError) return NextResponse.json({ error: relError.message, code: 'DB_ERROR' }, { status: 500 });
  }

  return NextResponse.json({ created: { persons: createdPersons.length, relationships: relInserts.length } }, { status: 201 });
}
```

```typescript
  "occurred_at": string           // ISO date YYYY-MM-DD, use today if ambiguous

  const user = await getAuthUser(supabase);
  if (!user) return NextResponse.json({ error: 'Unauthorized', code: 'AUTH_REQUIRED' }, { status: 401 });

  const body: unknown = await req.json();
  const parsed = ParseSchema.safeParse(body);
  if (!parsed.success) return NextResponse.json({ error: parsed.error.message, code: 'VALIDATION_ERROR' }, { status: 400 });

  const apiKey = process.env.GEMINI_API_KEY;
  if (!apiKey) return NextResponse.json({ error: 'Gemini not configured', code: 'GEMINI_ERROR' }, { status: 502 });

  try {
    const genAI = new GoogleGenerativeAI(apiKey);
    const model = genAI.getGenerativeModel({ model: 'gemini-1.5-flash' });

    const prompt = GEMINI_PROMPT.replace('{{TODAY}}', new Date().toISOString().slice(0, 10));
    const result = await model.generateContent(`${prompt}\n\nText: ${parsed.data.text}`);
    const raw = result.response.text().trim();

    // Parse and validate Gemini output
    let geminiData: {
      involved_persons: string[];
      event_type: string;
      description: string;
      impact: number;
      occurred_at: string;
    };
    try {
      geminiData = JSON.parse(raw);
    } catch {
      // Fallback if Gemini output is malformed
      geminiData = {
        involved_persons: [],
        event_type: 'note',
        description: parsed.data.text.slice(0, 100),
        impact: 0,
        occurred_at: new Date().toISOString().slice(0, 10),
      };
    }

    // Validate event_type
    if (!EVENT_TYPES.includes(geminiData.event_type as never)) geminiData.event_type = 'note';
    // Clamp impact
    geminiData.impact = Math.max(-3, Math.min(3, Math.round(geminiData.impact)));

    // Fuzzy match persons against DB
    // Use maybeSingle() — returns { data: null, error: null } when no row found (not an error)
    const matchedIds: (string | null)[] = await Promise.all(
      geminiData.involved_persons.map(async (name) => {
        const { data } = await supabase
          .from('persons')
          .select('id')
          .eq('user_id', user.id)
          .ilike('display_name', `%${name}%`)
          .limit(1)
          .maybeSingle();
        return data?.id ?? null;
      }),
    );

    return NextResponse.json({
      involved_persons: geminiData.involved_persons,
      matched_person_ids: matchedIds,
      event_type: geminiData.event_type,
      description: geminiData.description,
      impact: geminiData.impact,
      occurred_at: geminiData.occurred_at,
    });
  } catch {
    return NextResponse.json({ error: 'Gemini request failed', code: 'GEMINI_ERROR' }, { status: 502 });
  }
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { z } from 'zod';
import Anthropic from '@anthropic-ai/sdk';
import { createRouteClient, getAuthUser } from '@/lib/supabase-server';

const uuidFormat = z.string().regex(
  /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i,
);
const RequestSchema = z.object({
  from_id: uuidFormat,
  to_id:   uuidFormat,
});

const FREE_MONTHLY_LIMIT = 5;

async function checkAndIncrementUsage(
  supabase: Awaited<ReturnType<typeof createRouteClient>>,
  userId: string,
): Promise<boolean> {
  if (process.env.SKIP_AI_LIMIT === 'true') return true;

  const month = new Date().toISOString().slice(0, 7); // 'YYYY-MM'
  const { data } = await supabase
    .from('ai_usage')
    .select('count')
    .eq('user_id', userId)
    .eq('month', month)
    .single();

  const current = data?.count ?? 0;
  if (current >= FREE_MONTHLY_LIMIT) return false;

  await supabase.from('ai_usage').upsert(
    { user_id: userId, month, count: current + 1 },
    { onConflict: 'user_id,month' },
  );
  return true;
}

export async function POST(req: NextRequest) {
  const supabase = await createRouteClient();
  const user = await getAuthUser(supabase);
  if (!user) {
    return NextResponse.json({ error: 'Unauthorized', code: 'AUTH_REQUIRED' }, { status: 401 });
  }

  const body: unknown = await req.json();
  const parsed = RequestSchema.safeParse(body);
  if (!parsed.success) {
    return NextResponse.json({ error: 'Invalid request', code: 'VALIDATION_ERROR' }, { status: 400 });
  }

  const { from_id, to_id } = parsed.data;
  if (from_id === to_id) {
    return NextResponse.json({ error: '請選擇兩個不同的人', code: 'SAME_PERSON' }, { status: 400 });
  }

  // Get relationship paths (uses auth.uid() via RLS — no p_user_id param)
  const { data: paths, error: pathError } = await supabase.rpc('find_relationship_paths', {
    p_from_id: from_id,
    p_to_id: to_id,
    p_max_depth: 3,
  });

  if (pathError) {
    return NextResponse.json({ error: pathError.message, code: 'DB_ERROR' }, { status: 500 });
  }

  const pathList = paths ?? [];

  // No paths → skip AI
  if (pathList.length === 0) {
    return NextResponse.json({ paths: [], pathCount: 0, ai: null });
  }

  // Check usage limit
  const canUseAi = await checkAndIncrementUsage(supabase, user.id);
  if (!canUseAi) {
    return NextResponse.json({ paths: pathList, pathCount: pathList.length, ai: null, aiLimitReached: true });
  }

  // Resolve person names for paths
  const allIds = [...new Set(pathList.flatMap((p: { path: string[] }) => p.path))];
  const { data: personData } = await supabase
    .from('persons')
    .select('id, display_name')
    .in('id', allIds);

  const nameMap = new Map((personData ?? []).map((p: { id: string; display_name: string }) => [p.id, p.display_name]));

  const pathDescriptions = pathList.map((p: { path: string[]; min_closeness: number; depth: number }) => {
    const names = p.path.map((id: string) => nameMap.get(id) ?? '某人').join(' → ');
    return `路徑（${p.depth}度，最弱親密度 ${p.min_closeness}/5）：${names}`;
  }).join('\n');

  // Call Claude Sonnet
  const apiKey = process.env.ANTHROPIC_API_KEY;
  if (!apiKey) {
    return NextResponse.json({ paths: pathList, pathCount: pathList.length, ai: null });
  }

  try {
    const client = new Anthropic({ apiKey });
    const message = await client.messages.create({
      model: 'claude-sonnet-4-6',
      max_tokens: 400,
      messages: [{
        role: 'user',
        content: `你是一個社交風險分析助理。以下是兩人之間的關係路徑資料，請分析社交風險並給出建議。

關係路徑：
${pathDescriptions}

請以 JSON 格式回覆（不要有其他文字）：
{
  "riskLevel": "low" | "medium" | "high",
  "summary": "風險摘要（繁體中文，最多150字）",
  "suggestion": "建議（繁體中文，最多150字）"
}

規則：不做道德評判，不替用戶做決定，只提供分析和建議。`,
      }],
    });

    const aiText = message.content[0].type === 'text' ? message.content[0].text : '';
    let aiData: { riskLevel: 'low' | 'medium' | 'high'; summary: string; suggestion: string };
    try {
      aiData = JSON.parse(aiText);
      if (!['low', 'medium', 'high'].includes(aiData.riskLevel)) aiData.riskLevel = 'medium';
    } catch {
      aiData = { riskLevel: 'medium', summary: '無法解析分析結果', suggestion: '請自行判斷' };
    }

    return NextResponse.json({ paths: pathList, pathCount: pathList.length, ai: aiData });
  } catch {
    // Claude failure — return paths without AI
    return NextResponse.json({ paths: pathList, pathCount: pathList.length, ai: null });
  }
}
```
